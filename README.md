class Person {
private:
    // Private member variables
    std::string name;
    int age;

public:
    // Constructor
    Person(std::string n, int a) : name(n), age(a) {}

    // Public member function to display person details
    void displayInfo() {
        std::cout << "Name: " << name << "\nAge: " << age << std::endl;
    }

    // Getter for name
    std::string getName() {
        return name;
    }

    // Setter for name
    void setName(std::string n) {
        name = n;
    }

    // Getter for age
    int getAge() {
        return age;
    }

    // Setter for age
    void setAge(int a) {
        age = a;
    }
};
